\begin{tcolorbox}
	[colback=Emerald!10,colframe=cyan!40!black,title=\textbf{Question 24}]
	Translate in two ways each of these statements into logical expressions using predicates, quantifiers, and logical
	connectives. First, let the domain consist of the students
	in your class and second, let it consist of all people.
	\begin{enumerate}[\bfseries a)]
		\item Everyone in your class has a cellular phone.
		\item Somebody in your class has seen a foreign movie.
		\item There is a person in your class who cannot swim.
		\item All students in your class can solve quadratic equations.
		\item Some student in your class does not want to be rich.
	\end{enumerate}
\end{tcolorbox}
\begin{Sol}[]
	Let $C(x)$ be the propositional function ``x is in your class.''
	\begin{enumerate}[\bfseries a)]
		\item 
			Let $P(x)$ be the propositional function ``x has a cellular phone.''
			First we have $\forall xP(x)$, 
			and second we have $\forall x(C(x) \rightarrow P(x))$
		\item  
			Let $F(x)$ be the propositional function ``x has seen a foreign movie.''
			First we have $\exists xF(x)$, 
			and second we have $\exists x(C(x) \land F(x))$
		\item  
			Let $S(x)$ be the propositional function ``x cannot swim.''
			First we have $\exists x S(x)$, 
			and second we have $\exists x(C(x) \land S(x))$
		\item  
			Let $Q(x)$ be the propositional function ``x can solve quadratic equations.''
			First we have $\forall x Q(x)$, 
			and second we have $\forall x(C(x) \rightarrow Q(x))$
		\item  
			Let $R(x)$ be the propositional function ``x does not want to be rich.''
			First we have $\exists x R(x)$, 
			and second we have $\exists x(C(x) \land R(x))$
	\end{enumerate}
\end{Sol}

\begin{tcolorbox}
	[colback=Emerald!10,colframe=cyan!40!black,title=\textbf{Question 42}]
	Express each of these system specifications using predicates, quantifiers, and logical connectives.
	\begin{enumerate}[\bfseries a)]
		\item When there is less than 30 megabytes free on the hard
		disk, a warning message is sent to all users.
		\item No directories in the file system can be opened and
		no files can be closed when system errors have been
		detected.
		\item The file system cannot be backed up if there is a user
		currently logged on.
		\item Video on demand can be delivered when there are at
		least 8 megabytes of memory available and the connection speed is at least 56 kilobits per second.
	\end{enumerate}
\end{tcolorbox}
\begin{Sol}[]
	\begin{enumerate}[\bfseries a)]
		\item 
			Let $F(x)$ be a predicates that means ``There is less than $x$ megabytes free on the hard
			disk'', with the domain of discourse being positive numbers,
			and let $U(x)$ be ``User x is sent a warning message''. Then we can have$F(30) \rightarrow \forall x U(x)$
		\item 
			Let $S$ be ``System errors have been detected.'', and $D(x)$ be ``Directorie x in the file system can be opened''
			$I(x)$ be `` File x can be closed.''. Then we have $S \rightarrow \forall x \lnot D(x) \land \forall x \lnot F(x)$
		\item 
			Let $I$ be ``The file system can be backed up'', and $U(x)$ be ``User X currently logged on''
			Then we have $\exists x U(x) \rightarrow \lnot I$
		\item 
			Let $V$ be ``Video on demand can be delivered,'', and $M(x)$ be ``There are at least x megabytes of memory available''
			and let $C(x)$ be ``The connection speed is at least x kilobits per second''
			Then we have $(M(8) \land C(56)) \rightarrow V$
	\end{enumerate}
\end{Sol}

	Show that $\forall xP(x) \lor \forall xQ(x)$ and $\forall x(P(x) \lor Q(x))$ are not
	logically equivalent.
\end{tcolorbox}
\begin{Sol}[]
	Let $P(x)$ be true if and only if $x$ is odd, and let $Q(x)$ be true if and only if $x$ is even.
	Then for $\forall xP(x) \lor \forall xQ(x)$, it is false, while the proposition $\forall x(P(x) \lor Q(x))$ is true.
	Therefore, the two propositions are not logically equivalent.
\end{Sol}

		\item $\exists x\exists yP(x, y) \land \forall x\forall yQ(x, y)$
		\item $\exists x\exists y(Q(x, y) \leftrightarrow Q(y, x))$
		\item $\forall y\exists x\exists z(T(x, y, z) \lor Q(x, y))$
	\end{enumerate}
\end{tcolorbox}
\begin{Sol}[]
	\begin{enumerate}[\bfseries a)]
		\item $\forall z\exists y\exists x \lnot T(x, y, z)$
		\item $\forall x\forall yP(x, y) \lor \exists x\exists yQ(x, y)$
		\item $\forall x\forall y(Q(x, y) \leftrightarrow \lnot Q(y, x))$
		\item $\exists y\forall x\forall z(\lnot T(x, y, z) \land \lnot Q(x, y))$
	\end{enumerate}
\end{Sol}

\begin{tcolorbox}
	[colback=Emerald!10,colframe=cyan!40!black,title=\textbf{Question 6}]
	Use rules of inference to show that the hypotheses ``If it
	does not rain or if it is not foggy, then the sailing race will
	be held and the lifesaving demonstration will go on,'' ``If
	the sailing race is held, then the trophy will be awarded,''
	and ``The trophy was not awarded'' imply the conclusion
	``It rained.''
\end{tcolorbox}
\begin{Sol}[]
	Let proposition $r$ be ``It rains'', $f$ be ``it is foggy'', $s$ be ``the selling race will be held,'', $l$ be ``Lifesaving demonstration will go on,'' and $t$ be ``the trophy will be awarded''.
	Then we have the premises $\lnot r \lor \lnot f \rightarrow s \land l$, $s \rightarrow t$ and $\lnot t$. What we want to conclude is $r$ \\
	\begin{center}
		\begin{proof}
			\step $\lnot t$ & Hypothesis\\
			\step $s \rightarrow t$ & Hypothesis\\
			\step $\lnot s$ & Modus tollens using (1) and (2) \\
			\step $\lnot (s \land l)$ & Domination using (3) \\
			\step $\lnot r \land \lnot f \rightarrow s \land l$ & Hypothesis\\
			\step $\lnot (\lnot r \lor \lnot f)$ & Modus tollens using (4)\\
			\step $r \land f$ & De Morgan's law\\
			\step $r$ & Simplification using (7)\\
		\end{proof}
	\end{center}

\end{Sol}

		\item ``I am either dreaming or hallucinating.'' ``I am not

	\end{enumerate}
\end{tcolorbox}

		\item From ``All insects have six legs.'' and ``Dragonflies are insects.'', we can conclude that dragonflies are insects by using modus ponens, and we can also conclude that Spiders are not insects from ``Spiders eat dragonflies.'' by using modus tollens .

		\item From ``I am either dreaming or hallucinating.'' ``I am not
		dreaming.'', we can conclude that I am hallucinating by using disjunctive syllogism. And from ``If I am hallucinating, I see elephants running down the road.'' we can conclude that I  see elephants running down the road. by using modus ponens.
	\end{enumerate}
\end{Sol}

	Prove that if $n$ is a perfect square, then $n + 2$ is not a perfect square.
\end{tcolorbox}
\begin{Sol}[]
	If $n + 2$ is not a perfect square, supposed there exists integers $s^2=n$ and $t^2=n+2$, then we have $t^2 - s^2 = (t+s)(t-s)=2$. Because the factor of 2 is 1 and 2, $s+t = 2$ and $s-t=1$. We can not find such s and t. Therefore $n + 2$ is not a perfect square.
\end{Sol}

\end{Sol}

	[colback=Emerald!10,colframe=cyan!40!black,title=\textbf{Question 6}]
	Use a proof by cases to show that $min(a, min(b, c)) = min(min(a, b), c)$ whenever a, b, and c are real numbers.
\end{tcolorbox}
\begin{Sol}[]
	\begin{itemize}
		\item If a is smallest, then $a<min(b,c)$ and  $a<b \land a<c$, \\ so $min(a, min(b, c)) = min(min(a, b), c) = a$
		\item If b is smallest, then $b<c \land a>b$ and  $a>b \land b<c$, \\ so $min(a, min(b, c)) = min(min(a, b), c) = b$
		\item If c is smallest, then $b>c \land a>c$ and  $c<min(a,b)$, \\ so $min(a, min(b, c)) = min(min(a, b), c) = c$
	\end{itemize}
\end{Sol}

\end{Sol}
